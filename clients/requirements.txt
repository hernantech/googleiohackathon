# Device-simulator client deps — NOT orchestrator package deps.
# Install into the SAME venv (or a separate one):
#   .venv/bin/pip install -r clients/requirements.txt
#
# These pull in native audio/video stacks (PortAudio via sounddevice, the
# OpenCV runtime), which the orchestrator must never depend on — the server is
# device-blind and transcode-free. Hence a standalone requirements file.

numpy>=1.26         # audio buffers (sounddevice gives/takes numpy frames)
